			let g <- ( g - 5 * a ) * h;

			od;
			let g <- g * b / a + h
		else
			if c >= d then
				let e <- f * f * 7 - 2;
				while (d - 7) != e do
					let d <- d + a - 1;
					let e <- e - d + 1
				od;

					let d <- d - 1;
					let b <- b + 4;
					let e <- e + 1
				od;
				let f <- (g * f) / 4
			fi;
			let g <- g + h
		fi
